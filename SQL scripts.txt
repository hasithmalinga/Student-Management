CREATE TABLE `school_management`.`students` ( `id` INT NOT NULL AUTO_INCREMENT , `first_name` VARCHAR(255) NOT NULL , `last_name` VARCHAR(255) NOT NULL , `dob` DATE NOT NULL , `enrollment_date` DATE NOT NULL , `year` INT NOT NULL , `home_phone` VARCHAR(255) NOT NULL , `mobile` VARCHAR(255) NOT NULL , `email` VARCHAR(255) NOT NULL , `contact_name_1` VARCHAR(255) NOT NULL , `contact_phone_1` VARCHAR(255) NOT NULL , `contact_name_2` VARCHAR(255) NOT NULL , `contact_phone_2` VARCHAR(255) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
